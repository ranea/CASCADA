    def eval(cls, v0, v1, v2, v3):
        for i in range(cls.num_rounds):
            v0 += v1
            v1 = RotateLeft(v1, 5)
            v1 ^= v0
            v0 = RotateLeft(v0, 16)

            v2 += v3
            v3 = RotateLeft(v3, 8)
            v3 ^= v2

            v0 += v3
            v3 = RotateLeft(v3, 13)
            v3 ^= v0

            v2 += v1
            v1 = RotateLeft(v1, 7)
            v1 ^= v2
            v2 = RotateLeft(v2, 16)

            cls.add_round_outputs(v0, v1, v2, v3)

        return v0, v1, v2, v3
